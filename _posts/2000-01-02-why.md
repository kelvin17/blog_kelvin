##统计Latex的tex文件中中文字数脚本

# For paper

####使用时修改infiles指定的文件路径即可

	!/bin/env python
 	-*- coding: utf-8 -*-

	infiles = ["/Users/paper/绪论.tex",]

	def is_chinese(c):
    		if c  >= u'\u4e00' and c <= u'\u9fa5':
        		return True
		else:
			return False

	def count(infiles):
		total = 0
		for infile in infiles:
			c = 0
			for line in open(infile):
				line = line.decode("utf8")
			for each in line:
				if is_chinese(each):
				c += 1
			print "%s: %d" % (infile, c)
				total += c
			print "total count: %d" % (total)
		count(infiles)
